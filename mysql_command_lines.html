//create & drop a db
CREATE DATABASE databasename;
DROP DATABASE databasename;

//create a new table
CREATE TABLE Persons (
    PersonID int(8) NOT NULL AUTO_INCREMENT,
    LastName varchar(255),
    FirstName varchar(255) NOT NULL,
    userid varchar(50) COMMENT 'id of users saved here',
City varchar(255) DEFAULT 'DC',
UNIQUE (userid),
PRIMARY Key (PersonID)
);

//The following constraints are commonly used in SQL:

  • NOT NULL - Ensures that a column cannot have a NULL value
  • UNIQUE - Ensures that all values in a column are different
  • PRIMARY KEY - A combination of a NOT NULL and UNIQUE. Uniquely identifies each row in a table
  • FOREIGN KEY - Uniquely identifies a row/record in another table
  • CHECK - Ensures that all values in a column satisfies a specific condition
  • DEFAULT - Sets a default value for a column when no value is specified
  • INDEX - Use to create and retrieve data from the database very quickly

//alter a table=>add new column
ALTER TABLE table_name
ADD column_name datatype;

//alter a table=>remove a column
ALTER TABLE table_name
DROP COLUMN column_name;

//alter a table=> change a column
ALTER TABLE table_name
MODIFY COLUMN column_name datatype;

//Add an Index constraint to a table column
CREATE INDEX index_name
ON table_name (column1, column2, ...);

CREATE UNIQUE INDEX index_name
ON table_name (column1, column2, ...);

CREATE INDEX idx_lastname
ON Persons (LastName);

//creating a new table and adding a FOREIGN KEY or FK
CREATE TABLE Orders (
    OrderID int NOT NULL,
    OrderNumber int NOT NULL,
    PersonID int,
    PRIMARY KEY (OrderID),
    FOREIGN KEY (PersonID) REFERENCES Persons(PersonID)
);

//adding a FOREIGN KEY or FK to a table X, column Y by pointing it to its parent table Z and column W [after trable creation]
ALTER TABLE X ADD FOREIGN KEY (Y) REFERENCES Z(W);

//Dropping a FK
ALTER TABLE X
DROP FOREIGN KEY Y;
