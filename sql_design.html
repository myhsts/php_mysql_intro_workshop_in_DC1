MySQL Basic Queries

You need to download and import users.sql file to your db from below link:
https://github.com/myhsts/php_mysql_intro_workshop_in_DC1
Below queries are run on users database table:

//select one column
SELECT email FROM users

//select multiple columns
SELECT email, phone_no FROM users

//select all columns with filtering condition
SELECT * FROM users where pk>0

//sorting data DESC=====latest date show up first
SELECT * FROM users where pk>0 ORDER BY date_inserted DESC

//only seeing the first 3 records
SELECT * FROM users where pk>0 ORDER BY date_inserted DESC LIMIT 0,3

//adding As and mysql function: DATE_FORMAT
SELECT *, DATE_FORMAT(date_inserted,'%W %M %D') As nice_date FROM users where pk>0 ORDER BY date_inserted DESC LIMIT 0,3

//using IN function.....also, you can use NOT IN()
SELECT * FROM users where pk IN(1,2)

//using OR condition
SELECT * FROM users where (pk=3) OR (pk=4)

//using and condition
SELECT * FROM users where pk>0 AND name='matt'

//using BETWEEN
SELECT * FROM users where pk BETWEEN 1 AND 5

MySQL Advance Queries

//normal table join for one-to-many relationship
SELECT * FROM blog, pic where pk=parent_id

//normal table join for many-to-many relationship
SELECT * FROM blog, cat_info, blog_cat where pk=blog_id AND cat_pk=cat_id

//self-join [need to add join_id column with the same type as pk column)
SELECT * FROM blog as a, blog as b where a.pk=b.join_id

//Subqueries
SELECT * FROM blog WHERE pk <=9  AND pk >= (SELECT count(*)  FROM blog  WHERE title =  'matt'); 

//average, min (), max(), etc are like below:
SELECT AVG(column_name) FROM table_name

//Group by and Having [The HAVING clause was added to SQL because the WHERE keyword could not be used with aggregate functions]

SELECT column_name(s)
FROM table_name
WHERE condition
GROUP BY column_name(s)
HAVING condition
ORDER BY column_name(s);
